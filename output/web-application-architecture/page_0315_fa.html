Listing 12.2. Using this configuration deptrac can check if code follows the Dependency rule.


paths:
    - ./src
layers:
    - name: Infrastructure
      collectors:
          - type: className
            regex: .*\\Infrastructure\\.*
    - name: Domain
      collectors:
          - type: className
            regex: .*\\Domain\\.*
    - name: Application
      collectors:
          - type: className
            regex: .*\\Application\\.*
ruleset:
    Infrastructure:
        - Application
        - Domain
    Application:
        - Domain
    Domain:
        # nothing
